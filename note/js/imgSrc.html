  • img动态src问题

    # img 使用动态 src 遇到的问题

    # 示例 使用require

    // 无效示例 imgSrc 是全路径
const imgSrc = '@/assets/img/logo.png';
<img :src="getImgSrc()" />
function getImgSrc() {
  return require(imgSrc);
}

// 有效示例 
const img = 'logo.png';
<img :src="getImgSrc()" />

function getImgSrc() {
  return require('@/assets/img/' + img);
}
